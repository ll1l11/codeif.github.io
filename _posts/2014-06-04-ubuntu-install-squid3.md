运行命令:

    sudo apt-get install squid3

    p3@p3-ubuntu:/etc/squid3$ ls

    p3@p3-ubuntu:/etc/squid3$ whereis squid3
    squid3: /usr/sbin/squid3 /etc/squid3 /usr/lib/squid3 /usr/share/squid3 /usr/share/man/man8/squid3.8.gz
